52 }

54 bool merge(const PatternEnvironment& other)
55 {
56 for (const auto& [key, value] : other.bindings) {
57 if (bindings.contains(key)) {
58 // For variadic patterns, we want to accumulate values
59 bindings[key].insert(
60 bindings[key].end(),
61 value.begin(),
62 value.end());
63 } else {
64 bindings[key] = value;
65 }
66 }
67 return true;

157 std::cout << "\n=== Matching Variadic List ===\n";

209 return std::nullopt;

211 if (!env.merge(*subMatch)) {

213 return std::nullopt;
214 }
215 std::cout << "Successfully matched element " << i << "\n";

225class PatternSubstitutor {
226public:
227 std::shared_ptr<Expression> substitute(std::shared_ptr<Expression> templateExpr,
228 const PatternEnvironment& env)
229 {
230 if (!templateExpr)
231 return nullptr;
232
233 return std::visit(overloaded {
234 [&](const AtomExpression& atom) -> std::shared_ptr<Expression> {
235 // If atom is an identifier and exists in bindings, substitute it
236 if (atom.value.type == Tokentype::IDENTIFIER) {
237 auto it = env.bindings.find(atom.value.lexeme);
238 if (it != env.bindings.end() && !it->second.empty()) {
239 return it->second[0]; // Return first binding
240 }
241 }
242 // Otherwise return copy of original atom
243 return std::make_shared<Expression>(Expression { atom, templateExpr->line });
244 },
245
246 [&](const ListExpression& list) -> std::shared_ptr<Expression> {
247 std::vector<std::shared_ptr<Expression>> newElements;
248
249 // For each element in the template list
250 for (const auto& elem : list.elements) {
251 if (auto atomExpr = std::get_if<AtomExpression>(&elem->as)) {
252 // Check if it's a pattern variable
253 auto it = env.bindings.find(atomExpr->value.lexeme);
254 if (it != env.bindings.end()) {
255 // If variadic, expand all bindings
256 if (list.isVariadic) {
257 newElements.insert(newElements.end(),
258 it->second.begin(),
259 it->second.end());
260 continue;
261 }
262 }
263 }
264 // Regular substitution for non-variadic or non-pattern variables
265 auto substituted = substitute(elem, env);
266 if (substituted) {
267 newElements.push_back(substituted);
268 }
269 }
270
271 return std::make_shared<Expression>(
272 Expression { ListExpression { newElements, list.isVariadic }, templateExpr->line });
273 },
274
275 [&](const auto&) -> std::shared_ptr<Expression> {
276 return templateExpr;
277 } },
278 templateExpr->as);
279 }
